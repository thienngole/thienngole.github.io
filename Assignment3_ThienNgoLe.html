def acc_loss_graphic(model_history, title):
    fig = plt.figure()
    plt.subplot(2,1,2)
    plt.plot(model_history.history['acc'], label = 'trainning accuaracy')
    plt.plot(model_history.history['val_acc'], label = 'test accuaracy')
    plt.plot(model_history.history['loss'], label = 'trainning loss') 
    plt.plot(model_history.history['val_loss'], label = 'test loss')
    plt.title(title)
    plt.ylabel('loss/accuracy')
    plt.xlabel('epoch')
    plt.legend(bbox_to_anchor=(1,1), loc='upper left', ncol =1)

def classification_stat(model_name, model, x, y):
    
    predicted_classes = model.predict_classes(x)

    # see which we predicted correctly and which not
    correct_indices = np.nonzero(predicted_classes == y)[0]
    incorrect_indices = np.nonzero(predicted_classes != y)[0]
    print(model_name, " classification")
    print(len(correct_indices)," classified correctly")
    print(len(incorrect_indices)," classified incorrectly")
    
    plt.rcParams['figure.figsize'] = (7,14)
    figure_evaluation = plt.figure()
    print("Sample predictions.")
    for i, correct in enumerate(correct_indices[:3]):
        plt.subplot(6,3,i+1)
        plt.imshow(X_test[correct].reshape(28,28), cmap='gray', interpolation='none')
        plt.title("Predicted: {}, Truth: {}".format(predicted_classes[correct],
                                            y_test[correct]))
        plt.xticks([])
        plt.yticks([])
        
    for i, incorrect in enumerate(incorrect_indices[:3]):
        plt.subplot(6,3,i+4)
        plt.imshow(X_test[incorrect].reshape(28,28), cmap='gray', interpolation='none')
        plt.title("Predicted {}, Truth: {}".format(predicted_classes[incorrect], 
                                           y_test[incorrect]))
        plt.xticks([])
        plt.yticks([])
